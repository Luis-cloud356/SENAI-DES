DROP DATABASE if exists consultorio;
CREATE DATABASE consultorio;
USE consultorio;


CREATE TABLE pacientes (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nome VARCHAR(100),
  nascimento DATE
);

CREATE TABLE consultas (
  id INT AUTO_INCREMENT PRIMARY KEY,
  paciente_id INTEGER,
  data_consulta DATE,
  valor DECIMAL(10,2),
  FOREIGN KEY (paciente_id) REFERENCES pacientes(id)
);
